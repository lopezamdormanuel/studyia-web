document.getElementById('chatForm').addEventListener('submit', async (e) => { e.preventDefault(); const inputDelUsuario = document.getElementById('userInput'); const texto = inputDelUsuario.value.trim(); if (!texto) return; // 1. Limpiar el cuadro de texto para escribir la siguiente pregunta inputDelUsuario.value = ''; // 2. ¿El usuario escribió la palabra "dibuja"? if (texto.toLowerCase().startsWith('dibuja')) { try { // Le pedimos la imagen al servidor de Flask const response = await fetch('/generar-imagen', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: texto }) }); const data = await response.json(); // Mostramos la imagen en el chat mostrarImagenEnPantalla(data.url_imagen); } catch (error) { console.error('Error al crear la imagen:', error); } } else { // 3. Si NO dice "dibuja", es un texto normal (Como hablar con Gemini) try { const response = await fetch('/preguntar', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ mensaje: texto }) }); const data = await response.json(); mostrarTextoEnPantalla(data.respuesta); } catch (error) { console.error('Error en el chat de texto:', error); } } }); // ESTA FUNCIÓN PINTA LA IMAGEN EN TU CHATBOX function mostrarImagenEnPantalla(url) { const chatBox = document.getElementById('chatBox'); const mensajeBot = document.createElement('div'); mensajeBot.classList.add('message', 'bot'); const img = document.createElement('img'); img.src = url; img.style.maxWidth = '100%'; img.style.borderRadius = '8px'; img.style.marginTop = '10px'; mensajeBot.appendChild(img); chatBox.appendChild(mensajeBot); chatBox.scrollTop = chatBox.scrollHeight; } function mostrarTextoEnPantalla(texto) { const chatBox = document.getElementById('chatBox'); const mensajeBot = document.createElement('div'); mensajeBot.classList.add('message', 'bot'); mensajeBot.textContent = texto; chatBox.appendChild(mensajeBot); chatBox.scrollTop = chatBox.scrollHeight; }
